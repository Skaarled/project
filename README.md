string keyword (int level)
{
    char data[200];

    srand(time(NULL));
    int random=rand()%(3+3*(level-1))+(1+3*(level-1));
    ifstream infile;
    infile.open("keyword.txt");
    while (random--!=0)
        infile.getline(data,200);
    int i=0;
    while (data[i]!='$')
        {
            word=word+data[i];
            i++;
        }
    return word;
}

{

string toupperstr(string s)
{
    string str;
    for (int i=0;i<s.size();i++)
        if (s[i]!=' ')
                {
                    s[i]=toupper(s[i]);
                    str=str+s[i];
                }
    return str;
}
